160:      * Fire an event and call the listeners.

162:      * @param  string  $event
163:      * @param  mixed   $payload
164:      * @param  boolean $halt
165:      * @return void
166:      */
167:     public function fire($event, $payload = array(), $halt = false)
168:     {
169:         $responses = array();
170: 
171:         // If an array is not given to us as the payload, we will turn it into one so
172:         // we can easily use call_user_func_array on the listeners, passing in the
173:         // payload to each of them so that they receive each of these arguments.
174:         if ( ! is_array($payload)) $payload = array($payload);
175: 
176:         $payload[] = $event;
177: 
178:         foreach ($this->getListeners($event) as $listener)
179:         {
180:             $response = call_user_func_array($listener, $payload);
181: 
182:             // If a response is returned from the listener and event halting is enabled
183:             // we will just return this response, and not call the rest of the event
184:             // listeners. Otherwise we will add the response on the response list.
185:             if ( ! is_null($response) and $halt)
186:             {
187:                 return $response;
188:             }
189: 
190:             // If a boolean false is returned from a listener, we will stop propogating
191:             // the event to any further listeners down in the chain, else we keep on
192:             // looping through the listeners and firing every one in our sequence.
193:             if ($response === false) break;
194: 
195:             $responses[] = $response;
196:         }
197: 
198:         return $halt ? null : $responses;
199:     }

276:      *
277:      * @param  mixed    $listener
278:      * @return Closure
279:      */
280:     public function createClassListener($listener)
281:     {
282:         $container = $this->container;
283: 
284:         return function() use ($listener, $container)
285:         {
286:             // If the listener has an @ sign, we will assume it is being used to delimit
287:             // the class name from the handle method name. This allows for handlers
288:             // to run multiple handler methods in a single class for convenience.
289:             $segments = explode('@', $listener);
290: 
291:             $method = count($segments) == 2 ? $segments[1] : 'handle';
292: 
293:             $callable = array($container->make($segments[0]), $method);
294: 
295:             // We will make a callable of the listener instance and a method that should
296:             // be called on that instance, then we will pass in the arguments that we
297:             // received in this method into this listener class instance's methods.
298:             $data = func_get_args();
299: 
300:             return call_user_func_array($callable, $data);
301:         };
302:     }
