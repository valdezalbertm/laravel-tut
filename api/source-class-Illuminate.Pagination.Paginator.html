135:     protected function calculateCurrentPage($lastPage)
136:     {
137:         $page = $this->env->getCurrentPage();
138: 
139:         // The page number will get validated and adjusted if it either less than one
140:         // or greater than the last page available based on the count of the given
141:         // items array. If it's greater than the last, we'll give back the last.
142:         if (is_numeric($page) and $page > $lastPage)
143:         {
144:             return $lastPage > 0 ? $lastPage : 1;
145:         }
