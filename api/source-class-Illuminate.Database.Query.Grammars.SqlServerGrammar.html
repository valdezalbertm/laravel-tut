 21:     {
 22:         $components = $this->compileComponents($query);
 23: 
 24:         // If an offset is present on the query, we will need to wrap the query in

 26:         // other database systems but is necessary for implementing features.

 36:      * Compile the "select *" portion of the query.
 37:      *
 38:      * @param  \Illuminate\Database\Query\Builder  $query
 39:      * @param  array  $columns
 40:      * @return string
 41:      */
 42:     protected function compileColumns(Builder $query, $columns)
 43:     {
 44:         if ( ! is_null($query->aggregate)) return;
 45: 
 46:         $select = $query->distinct ? 'select distinct ' : 'select ';
 47: 
 48:         // If there is a limit on the query, but not an offset, we will add the top
 49:         // clause to the query, which serves as a "limit" type clause within the
 50:         // SQL Server system similar to the limit keywords available in MySQL.
 51:         if ($query->limit > 0 and $query->offset <= 0)
 52:         {
 53:             $select .= 'top '.$query->limit.' ';
 54:         }
 55: 
 56:         return $select.$this->columnize($columns);
 57:     }

 59:     /**
 60:      * Create a full ANSI offset clause for the query.
 61:      *
 62:      * @param  \Illuminate\Database\Query\Builder  $query
 63:      * @param  array  $components
 64:      * @return string
 65:      */
 66:     protected function compileAnsiOffset(Builder $query, $components)
 67:     {
 68:         // An ORDER BY clause is required to make this offset query work, so if one does
 69:         // not exist we'll just create a dummy clause to trick the database and so it
 70:         // does not complain about the queries for not having an "order by" clause.
 71:         if ( ! isset($components['orders']))
 72:         {
 73:             $components['orders'] = 'order by (select 0)';
 74:         }
 75: 
 76:         // We need to add the row number to the query so we can compare it to the offset
 77:         // and limit values given for the statements. So we will add an expression to
 78:         // the "select" that will give back the row numbers on each of the records.
 79:         $orderings = $components['orders'];
 80: 
 81:         $components['columns'] .= $this->compileOver($orderings);
 82: 
 83:         unset($components['orders']);
 84: 
 85:         // Next we need to calculate the constraints that should be placed on the query
 86:         // to get the right offset and limit from our query but if there is no limit
 87:         // set we will just handle the offset only since that is all that matters.
 88:         $start = $query->offset + 1;
 89: 
 90:         $constraint = $this->compileRowConstraint($query);
 91: 
 92:         $sql = $this->concatenate($components);
 93: 
 94:         // We are now ready to build the final SQL query so we'll create a common table
 95:         // expression from the query and get the records with row numbers within our
 96:         // given limit and offset value that we just put on as a query constraint.
 97:         return $this->compileTableExpression($sql, $constraint);
 98:     }
