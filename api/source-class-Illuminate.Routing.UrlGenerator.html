 85:     {
 86:         if ($this->isValidUrl($path)) return $path;
 87: 
 88:         $scheme = $this->getScheme($secure);

206:     /**
207:      * Build the parameter list for short circuit parameters.
208:      *
209:      * @param  \Illuminate\Routing\Route  $route
210:      * @param  array  $params
211:      * @return array
212:      */
213:     protected function buildParameterList($route, array $params)
214:     {
215:         $keys = $route->getParameterKeys();
216: 
217:         // If the number of keys is less than the number of parameters on a route
218:         // we'll fill out the parameter arrays with empty bindings on the rest
219:         // of the spots until they are equal so we can run an array combine.
220:         if (count($params) < count($keys))
221:         {
222:             $difference = count($keys) - count($params);
223: 
224:             $params += array_fill(count($params), $difference, null);
225:         }
226: 
227:         return array_combine($keys, $params);
228:     }

230:     /**
231:      * Get the URL to a controller action.
232:      *
233:      * @param  string  $action
234:      * @param  mixed   $parameters
235:      * @param  bool    $absolute
236:      * @return string
237:      */
238:     public function action($action, $parameters = array(), $absolute = true)
239:     {
240:         // First we'll check to see if we have already rendered a URL for an action
241:         // so that we don't have to loop through all of the routes again on each
242:         // iteration through the loop. If we have it, we can just return that.
243:         if (isset($this->actionMap[$action]))
244:         {
245:             $name = $this->actionMap[$action];
246: 
247:             return $this->route($name, $parameters, $absolute);
248:         }
249: 
250:         // If haven't already mapped this action to a URI yet, we will need to spin
251:         // through all of the routes looking for routes that routes to the given
252:         // controller's action, then we will cache them off and build the URL.
253:         foreach ($this->routes as $name => $route)
254:         {
255:             if ($action == $route->getOption('_uses'))
256:             {
257:                 $this->actionMap[$action] = $name;
258: 
259:                 return $this->route($name, $parameters, $absolute);
260:             }
261:         }
262: 
263:         throw new InvalidArgumentException("Unknown action [$action].");
264:     }
