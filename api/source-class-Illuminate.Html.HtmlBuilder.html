328:     public function attributes($attributes)
329:     {

332:         // For numeric keys we will assume that the key and the value are the same
333:         // as this will convert HTML attributes such as "required" to a correct

360:      * Obfuscate a string to prevent spam-bots from sniffing it.
361:      *
362:      * @param  string  $value
363:      * @return string
364:      */
365:     public function obfuscate($value)
366:     {
367:         $safe = '';
368: 
369:         foreach (str_split($value) as $letter)
370:         {
371:             // To properly obfuscate the value, we will randomly convert each letter to
372:             // its entity or hexadecimal representation, keeping a bot from sniffing
373:             // the randomly obfuscated letters out of the string on the responses.
374:             switch (rand(1, 3))
375:             {
376:                 case 1:
377:                     $safe .= '&#'.ord($letter).';'; break;
378: 
379:                 case 2:
380:                     $safe .= '&#x'.dechex(ord($letter)).';'; break;
381: 
382:                 case 3:
383:                     $safe .= $letter;
384:             }
385:         }
386: 
387:         return $safe;
388:     }
