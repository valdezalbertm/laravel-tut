 73: 
 74:         // We will keep track of the amount of views being rendered so we can flush
 75:         // the section after the complete rendering operation is done. This will
 76:         // clear out the sections for any separate views that may be rendered.
 77:         $env->incrementRender();
 78: 
 79:         $env->callComposer($this);
 80: 
 81:         $contents = trim($this->getContents());
 82: 
 83:         // Once we've finished rendering the view, we'll decrement the render count

 85:         // they might interfere with totally separate view's evaluations later.

 88:         if ($env->doneRendering()) $env->flushSections();
