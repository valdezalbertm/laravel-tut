212:     /**
213:      * Handle the given exception.
214:      *
215:      * @param  Exception  $exception
216:      * @param  bool  $fromConsole
217:      * @return void
218:      */
219:     protected function callCustomHandlers($exception, $fromConsole = false)
220:     {
221:         foreach ($this->handlers as $handler)
222:         {
223:             // If this exception handler does not handle the given exception, we will just
224:             // go the next one. A handler may type-hint an exception that it handles so
225:             //  we can have more granularity on the error handling for the developer.
226:             if ( ! $this->handlesException($handler, $exception))
227:             {
228:                 continue;
229:             }
230:             elseif ($exception instanceof HttpExceptionInterface)
231:             {
232:                 $code = $exception->getStatusCode();
233:             }
234: 
235:             // If the exception doesn't implement the HttpExceptionInterface, we will just
236:             // use the generic 500 error code for a server side error. If it implements
237:             // the HttpException interfaces we'll grab the error code from the class.
238:             else
239:             {
240:                 $code = 500;
241:             }
242: 
243:             // We will wrap this handler in a try / catch and avoid white screens of death
244:             // if any exceptions are thrown from a handler itself. This way we will get
245:             // at least some errors, and avoid errors with no data or not log writes.
246:             try
247:             {
248:                 $response = $handler($exception, $code, $fromConsole);
249:             }
250:             catch (\Exception $e)
251:             {
252:                 $response = $this->formatException($e);
253:             }
254: 
255:             // If this handler returns a "non-null" response, we will return it so it will
256:             // get sent back to the browsers. Once the handler returns a valid response
257:             // we will cease iterating through them and calling these other handlers.
258:             if (isset($response) and ! is_null($response))
259:             {
260:                 return $response;
261:             }
262:         }
263:     }
