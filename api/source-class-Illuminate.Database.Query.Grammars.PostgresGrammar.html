 15:     {
 16:         $table = $this->wrapTable($query->from);

 18:         // Each one of the columns in the update statements needs to be wrapped in the
 19:         // keyword identifiers, also a place-holder needs to be created for each of
 20:         // the values in the list of bindings so we can make the sets statements.
 21:         $columns = $this->compileUpdateColumns($values);

 26:

 31:      * Compile the columns for the update statement.
 32:      *
 33:      * @param  array   $values
 34:      * @return string
 35:      */
 36:     protected function compileUpdateColumns($values)
 37:     {
 38:         $columns = array();
 39: 
 40:         // When gathering the columns for an update statement, we'll wrap each of the
 41:         // columns and convert it to a parameter value. Then we will concatenate a
 42:         // list of the columns that can be added into this update query clauses.
 43:         foreach ($values as $key => $value)
 44:         {
 45:             $columns[] = $this->wrap($key).' = '.$this->parameter($value);
 46:         }
 47: 
 48:         return implode(', ', $columns);
 49:     }

 51:     /**
 52:      * Compile the "from" clause for an update with a join.
 53:      *
 54:      * @param  \Illuminate\Database\Query\Builder  $query
 55:      * @return string
 56:      */
 57:     protected function compileUpdateFrom(Builder $query)
 58:     {
 59:         if ( ! isset($query->joins)) return '';
 60: 
 61:         $froms = array();
 62: 
 63:         // When using Postgres, updates with joins list the joined tables in the from
 64:         // clause, which is different than other systems like MySQL. Here, we will
 65:         // compile out the tables that are joined and add them to a from clause.
 66:         foreach ($query->joins as $join)
 67:         {
 68:             $froms[] = $this->wrapTable($join->table);
 69:         }
 70: 
 71:         if (count($froms) > 0) return ' from '.implode(', ', $froms);
 72:     }

 74:     /**
 75:      * Compile the additional where clauses for updates with joins.
 76:      *
 77:      * @param  \Illuminate\Database\Query\Builder  $query
 78:      * @return string
 79:      */
 80:     protected function compileUpdateWheres(Builder $query)
 81:     {
 82:         $baseWhere = $this->compileWheres($query);
 83: 
 84:         if ( ! isset($query->joins)) return $baseWhere;
 85: 
 86:         // Once we compile the join constraints, we will either use them as the where
 87:         // clause or append them to the existing base where clauses. If we need to
 88:         // strip the leading boolean we will do so when using as the only where.
 89:         $joinWhere = $this->compileUpdateJoinWheres($query);
 90: 
 91:         if (trim($baseWhere) == '')
 92:         {
 93:             return 'where '.$this->removeLeadingBoolean($joinWhere);
 94:         }
 95:         else
 96:         {
 97:             return $baseWhere.' '.$joinWhere;
 98:         }
 99:     }

111:         // Here we will just loop through all of the join constraints and compile them
