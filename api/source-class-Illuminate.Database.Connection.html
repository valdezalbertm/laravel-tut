368:      * Prepare the query bindings for execution.
369:      *
370:      * @param  array  $bindings
371:      * @return array
372:      */
373:     public function prepareBindings(array $bindings)
374:     {
375:         $grammar = $this->getQueryGrammar();
376: 
377:         foreach ($bindings as $key => $value)
378:         {
379:             // We need to transform all instances of the DateTime class into an actual
380:             // date string. Each query grammar maintains its own date string format
381:             // so we'll just ask the grammar for the format to get from the date.
382:             if ($value instanceof DateTime)
383:             {
384:                 $bindings[$key] = $value->format($grammar->getDateFormat());
385:             }
386:             elseif ($value === false)
387:             {
388:                 $bindings[$key] = 0;
389:             }
390:         }
391: 
392:         return $bindings;
393:     }

395:     /**
396:      * Execute a Closure within a transaction.
397:      *
398:      * @param  Closure  $callback
399:      * @return mixed
400:      */
401:     public function transaction(Closure $callback)
402:     {
403:         $this->pdo->beginTransaction();
404: 
405:         // We'll simply execute the given callback within a try / catch block
406:         // and if we catch any exception we can rollback the transaction
407:         // so that none of the changes are persisted to the database.
408:         try
409:         {
410:             $result = $callback($this);
411: 
412:             $this->pdo->commit();
413:         }
414: 
415:         // If we catch an exception, we will roll back so nothing gets messed
416:         // up in the database. Then we'll re-throw the exception so it can
417:         // be handled how the developer sees fit for their applications.
418:         catch (\Exception $e)
419:         {
420:             $this->pdo->rollBack();
421: 
422:             throw $e;
423:         }
424: 
425:         return $result;
426:     }

450:     /**
451:      * Run a SQL statement and log its execution context.
452:      *
453:      * @param  string   $query
454:      * @param  array    $bindings
455:      * @param  Closure  $callback
456:      * @return mixed
457:      */
458:     protected function run($query, $bindings, Closure $callback)
459:     {
460:         $start = microtime(true);
461: 
462:         // To execute the statement, we'll simply call the callback, which will actually
463:         // run the SQL against the PDO connection. Then we can calculate the time it
464:         // took to execute and log the query SQL, bindings and time in our memory.
465:         try
466:         {
467:             $result = $callback($this, $query, $bindings);
468:         }
469: 
470:         // If an exception occurs when attempting to run a query, we'll format the error
471:         // message to include the bindings with SQL, which will make this exception a
472:         // lot more helpful to the developer instead of just the database's errors.
473:         catch (\Exception $e)
474:         {
475:             $this->handleQueryException($e, $query, $bindings);
476:         }
477: 
478:         // Once we have run the query we will calculate the time that it took to run and
479:         // then log the query, bindings, and execution time so we will report them on
480:         // the event that the developer needs them. We'll log time in milliseconds.
481:         $time = $this->getElapsedTime($start);
482: 
483:         $this->logQuery($query, $bindings, $time);
484: 
485:         return $result;
486:     }
