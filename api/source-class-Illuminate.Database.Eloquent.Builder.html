408:      *
409:      * @param  array  $models
410:      * @return array

412:     public function eagerLoadRelations(array $models)
413:     {
414:         foreach ($this->eagerLoad as $name => $constraints)
415:         {
416:             // For nested eager loads we'll skip loading them here and they will be set as an
417:             // eager load on the query to retrieve the relation so that they will be eager
418:             // loaded on that query, because that is where they get hydrated as models.
419:             if (strpos($name, '.') === false)
420:             {

422:             }
423:         }
424: 
425:         return $models;

428:     /**
429:      * Eagerly load the relationship on a set of models.
430:      *
431:      * @param  string   $relation
432:      * @param  array    $models
433:      * @param  Closure  $constraints
434:      * @return array
435:      */
436:     protected function loadRelation(array $models, $name, Closure $constraints)
437:     {
438:         // First we will "back up" the existing where conditions on the query so we can
439:         // add our eager constraints. Then we will merge the wheres that were on the
440:         // query back to it in order that any where conditions might be specified.
441:         $relation = $this->getRelation($name);
442: 
443:         list($wheres, $bindings) = $relation->getAndResetWheres();
444: 
445:         $relation->addEagerConstraints($models);
446: 
447:         // We allow the developers to specify constraints on eager loads and we'll just
448:         // call the constraints Closure, passing along the query so they will simply
449:         // do all they need to the queries, and even may specify non-where things.
450:         $relation->mergeWheres($wheres, $bindings);
451: 
452:         call_user_func($constraints, $relation);
453: 
454:         $models = $relation->initRelation($models, $name);
455: 
456:         // Once we have the results, we just match those back up to their parent models
457:         // using the relationship instance. Then we just return the finished arrays
458:         // of models which have been eagerly hydrated and are readied for return.
459:         $results = $relation->get();
460: 
461:         return $relation->match($models, $results, $name);
462:     }

575:     /**
576:      * Parse a list of relations into individuals.
577:      *
578:      * @param  array  $relations
579:      * @return array
580:      */
581:     protected function parseRelations(array $relations)
582:     {
583:         $results = array();
584: 
585:         foreach ($relations as $name => $constraints)
586:         {
587:             // If the "relation" value is actually a numeric key, we can assume that no
588:             // constraints have been specified for the eager load and we'll just put
589:             // an empty Closure with the loader so that we can treat all the same.
590:             if (is_numeric($name))
591:             {
592:                 $f = function() {};
593: 
594:                 list($name, $constraints) = array($constraints, $f);
595:             }
596: 
597:             // We need to separate out any nested includes. Which allows the developers
598:             // to load deep relationships using "dots" without stating each level of
599:             // the relationship with its own key in the array of eager load names.
600:             $results = $this->parseNested($name, $results);
601: 
602:             $results[$name] = $constraints;
603:         }
604: 
605:         return $results;
606:     }

608:     /**
609:      * Parse the nested relationships in a relation.
610:      *
611:      * @param  string  $name
612:      * @param  array   $results
613:      * @return array
614:      */
615:     protected function parseNested($name, $results)
616:     {
617:         $progress = array();
618: 
619:         // If the relation has already been set on the result array, we will not set it
620:         // again, since that would override any constraints that were already placed
621:         // on the relationships. We will only set the ones that are not specified.
622:         foreach (explode('.', $name) as $segment)
623:         {
624:             $progress[] = $segment;
625: 
626:             if ( ! isset($results[$last = implode('.', $progress)]))
627:             {
628:                 $results[$last] = function() {};
629:             }
630:         }
631: 
632:         return $results;
633:     }
