452:     /**
453:      * Sync the intermediate tables with a list of IDs.
454:      *
455:      * @param  array  $ids
456:      * @return void
457:      */
458:     public function sync(array $ids)
459:     {
460:         // First we need to attach any of the associated models that are not currently
461:         // in this joining table. We'll spin through the given IDs, checking to see
462:         // if they exist in the array of current ones, and if not we will insert.
463:         $current = $this->newPivotQuery()->lists($this->otherKey);
464: 
465:         $records = $this->formatSyncList($ids);
466: 
467:         $detach = array_diff($current, array_keys($records));
468: 
469:         // Next, we will take the differences of the currents and given IDs and detach
470:         // all of the entities that exist in the "current" array but are not in the
471:         // the array of the IDs given to the method which will complete the sync.
472:         if (count($detach) > 0)
473:         {
474:             $this->detach($detach);
475:         }
476: 
477:         // Now we are finally ready to attach the new records. Note that we'll disable
478:         // touching until after the entire operation is complete so we don't fire a
479:         // ton of touch operations until we are totally done syncing the records.
480:         $this->attachNew($records, $current, false);
481: 
482:         $this->touchIfTouching();
483:     }

674:      * Detach models from the relationship.
675:      *
676:      * @param  int|array  $ids
677:      * @param  bool  $touch
678:      * @return int
679:      */
680:     public function detach($ids = array(), $touch = true)
681:     {
682:         if ($ids instanceof Model) $ids = (array) $ids->getKey();
683: 
684:         $query = $this->newPivotQuery();
685: 
686:         // If associated IDs were passed to the method we will only delete those
687:         // associations, otherwise all of the association ties will be broken.
688:         // We'll return the numbers of affected rows when we do the deletes.
689:         $ids = (array) $ids;
690: 
691:         if (count($ids) > 0)
692:         {
693:             $query->whereIn($this->otherKey, $ids);
694:         }
695: 
696:         if ($touch) $this->touchIfTouching();
697: 
698:         // Once we have all of the conditions set on the statement, we are ready
