 52:         }
 53: 
 54:         // If the application is running in the console, we will not lazy load any of
 55:         // the service providers. This is mainly because it's not as necessary for

 57:         if ($app->runningInConsole())
 58:         {
 59:             $manifest['eager'] = $manifest['providers'];
 60:         }
 61:

 63:         // application so their services can be registered with the application as

 65:         foreach ($manifest['eager'] as $provider)
 66:         {
 67:             $app->register($this->createProvider($app, $provider));
 68:         }

 71:     }

 74:      * Compile the application manifest file.

 77:      * @param  array  $providers
 78:      * @return array
 79:      */
 80:     protected function compileManifest(Application $app, $providers)
 81:     {
 82:         // The service manifest should contain a list of all of the providers for
 83:         // the application so we can compare it on each request to the service
 84:         // and determine if the manifest should be recompiled or is current.
 85:         $manifest = $this->freshManifest($providers);
 86: 
 87:         foreach ($providers as $provider)
 88:         {
 89:             $instance = $this->createProvider($app, $provider);
 90: 
 91:             // When recompiling the service manifest, we will spin through each of the
 92:             // providers and check if it's a deferred provider or not. If so we'll
 93:             // add it's provided services to the manifest and note the provider.
 94:             if ($instance->isDeferred())
 95:             {
 96:                 foreach ($instance->provides() as $service)
 97:                 {
 98:                     $manifest['deferred'][$service] = $provider;
 99:                 }
100:             }
101: 
102:             // If the service providers are not deferred, we will simply add it to an
103:             // of eagerly loaded providers that will be registered with the app on
104:             // each request to the applications instead of being lazy loaded in.
105:             else

108:             }
109:         }

112:     }
