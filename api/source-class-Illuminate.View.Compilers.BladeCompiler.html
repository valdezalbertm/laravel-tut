113:     /**
114:      * Compile Blade template extensions into valid PHP.
115:      *
116:      * @param  string  $value
117:      * @return string
118:      */
119:     protected function compileExtends($value)
120:     {
121:         // By convention, Blade views using template inheritance must begin with the
122:         // @extends expression, otherwise they will not be compiled with template
123:         // inheritance. So, if they do not start with that we will just return.
124:         if (strpos($value, '@extends') !== 0)
125:         {
126:             return $value;
127:         }
128: 
129:         $lines = preg_split("/(\r?\n)/", $value);
130: 
131:         // Next, we just want to split the values by lines, and create an expression
132:         // to include the parent layout at the end of the templates. Which allows
133:         // the sections to get registered before the parent view gets rendered.
134:         $lines = $this->compileLayoutExtends($lines);
135: 
136:         return implode("\r\n", array_slice($lines, 1));
137:     }
