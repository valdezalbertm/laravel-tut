161:     /**
162:      * Parse a filter name and add any parameters to the array.
163:      *
164:      * @param  string  $name
165:      * @param  array   $parameters
166:      * @return array
167:      */
168:     protected function parseFilter($name, $parameters = array())
169:     {
170:         if (str_contains($name, ':'))
171:         {
172:             // If the filter name contains a colon, we will assume that the developer
173:             // is passing along some parameters with the name, and we will explode
174:             // out the name and paramters, merging the parameters onto the list.
175:             $segments = explode(':', $name);
176: 
177:             $name = $segments[0];
178: 
179:             // We will merge the arguments specified in the filter name into the list
180:             // of existing parameters. We'll send them at the end since any values
181:             // at the front are usually static such as request, response, route.
182:             $arguments = explode(',', $segments[1]);
183: 
184:             $parameters = array_merge($parameters, $arguments);
185:         }
186: 
187:         return array($name, $parameters);
188:     }
