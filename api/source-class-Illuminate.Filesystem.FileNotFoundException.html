280:     /**
281:      * Copy a directory from one location to another.
282:      *
283:      * @param  string  $directory
284:      * @param  string  $destination
285:      * @param  int     $options
286:      * @return void
287:      */
288:     public function copyDirectory($directory, $destination, $options = null)
289:     {
290:         if ( ! $this->isDirectory($directory)) return false;
291: 
292:         $options = $options ?: FilesystemIterator::SKIP_DOTS;
293: 
294:         // If the destination directory does not actually exist, we will go ahead and
295:         // create it recursively, which just gets the destination prepared to copy
296:         // the files over. Once we make the directory we'll proceed the copying.
297:         if ( ! $this->isDirectory($destination))
298:         {
299:             $this->makeDirectory($destination, 0777, true);
300:         }
301: 
302:         $items = new FilesystemIterator($directory, $options);
303: 
304:         foreach ($items as $item)
305:         {
306:             // As we spin through items, we will check to see if the current file is actually
307:             // a directory or a file. When it is actually a directory we will need to call
308:             // back into this function recursively to keep copying these nested folders.
309:             $target = $destination.'/'.$item->getBasename();
310: 
311:             if ($item->isDir())
312:             {
313:                 $path = $item->getRealPath();
314: 
315:                 if ( ! $this->copyDirectory($path, $target, $options)) return false;
316:             }
317: 
318:             // If the current items is just a regular file, we will just copy this to the new
319:             // location and keep looping. If for some reason the copy fails we'll bail out
320:             // and return false, so the developer is aware that the copy process failed.
321:             else
322:             {
323:                 if ( ! $this->copy($item->getRealPath(), $target)) return false;
324:             }
325:         }
326: 
327:         return true;
328:     }

331:      * Recursively delete a directory.
332:      *
333:      * The directory itself may be optionally preserved.
334:      *
335:      * @param  string  $directory
336:      * @param  bool    $preserve
337:      * @return void
338:      */
339:     public function deleteDirectory($directory, $preserve = false)
340:     {
341:         if ( ! $this->isDirectory($directory)) return;
342: 
343:         $items = new FilesystemIterator($directory);
344: 
345:         foreach ($items as $item)
346:         {
347:             // If the item is a directory, we can just recurse into the function and
348:             // delete that sub-director, otherwise we'll just delete the file and
349:             // keep iterating through each file until the directory is cleaned.
350:             if ($item->isDir())
351:             {
352:                 $this->deleteDirectory($item->getRealPath());
353:             }
354: 
355:             // If the item is just a file, we can go ahead and delete it since we're
356:             // just looping through and waxing all of the files in this directory
357:             // and calling directories recursively, so we delete the real path.
358:             else
359:             {
360:                 $this->delete($item->getRealPath());
361:             }
362:         }
363: 
364:         if ( ! $preserve) @rmdir($directory);
365:     }
