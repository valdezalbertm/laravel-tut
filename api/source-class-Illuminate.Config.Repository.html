142:         $env = $this->environment;
143: 
144:         // If we've already loaded this collection, we will just bail out since we do
145:         // not want to load it again. Once items are loaded a first time they will
146:         // stay kept in memory within this class and not loaded from disk again.
147:         if (isset($this->items[$collection]))

154:         // If we've already loaded this collection, we will just bail out since we do
155:         // not want to load it again. Once items are loaded a first time they will
