219:     /**
220:      * Set the application environment for a web request.
221:      *
222:      * @param  string  $base
223:      * @param  array|string  $environments
224:      * @return string
225:      */
226:     protected function detectWebEnvironment($base, $environments)
227:     {
228:         // If the given environment is just a Closure, we will defer the environment
229:         // detection to the Closure the developer has provided, which allows them
230:         // to totally control the web environment detection if they require to.
231:         if ($environments instanceof Closure)
232:         {
233:             return $this['env'] = call_user_func($environments);
234:         }
235: 
236:         foreach ($environments as $environment => $hosts)
237:         {
238:             // To determine the current environment, we'll simply iterate through the
239:             // possible environments and look for a host that matches this host in
240:             // the request's context, then return back that environment's names.
241:             foreach ((array) $hosts as $host)
242:             {
243:                 if (str_is($host, $base) or $this->isMachine($host))
244:                 {
245:                     return $this['env'] = $environment;
246:                 }
247:             }
248:         }
249: 
250:         return $this['env'] = 'production';
251:     }
