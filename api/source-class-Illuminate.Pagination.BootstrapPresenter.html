 69:     {
 70:         $pages = array();
 71: 
 72:         for ($page = $start; $page <= $end; $page++)
 73:         {
 74:             // If the current page is equal to the page we're iterating on, we will create a

 77:             if ($this->currentPage == $page)
 78:             {

 80:             }
 81:             else
 82:             {

 90:     /**
 91:      * Create a pagination slider link window.
 92:      *
 93:      * @return string
 94:      */
 95:     protected function getPageSlider()
 96:     {
 97:         $window = 6;
 98: 
 99:         // If the current page is very close to the beginning of the page range, we will
100:         // just render the beginning of the page range, followed by the last 2 of the
101:         // links in this list, since we will not have room to create a full slider.
102:         if ($this->currentPage <= $window)
103:         {
104:             $ending = $this->getFinish();
105: 
106:             return $this->getPageRange(1, $window + 2).$ending;
107:         }
108: 
109:         // If the current page is close to the ending of the page range we will just get
110:         // this first couple pages, followed by a larger window of these ending pages
111:         // since we're too close to the end of the list to create a full on slider.
112:         elseif ($this->currentPage >= $this->lastPage - $window)
113:         {
114:             $start = $this->lastPage - 8;
115: 
116:             $content = $this->getPageRange($start, $this->lastPage);
117: 
118:             return $this->getStart().$content;
119:         }
120: 
121:         // If we have enough room on both sides of the current page to build a slider we
122:         // will surround it with both the beginning and ending caps, with this window
123:         // of pages in the middle providing a Google style sliding paginator setup.
124:         else
125:         {
126:             $content = $this->getAdjacentRange();

128:             return $this->getStart().$content.$this->getFinish();
129:         }
