 49:      * Load the given configuration group.
 50:      *
 51:      * @param  string  $environment
 52:      * @param  string  $group
 53:      * @param  string  $namespace
 54:      * @return array
 55:      */
 56:     public function load($environment, $group, $namespace = null)
 57:     {
 58:         $items = array();
 59: 
 60:         // First we'll get the root configuration path for the environment which is
 61:         // where all of the configuration files live for that namespace, as well
 62:         // as any environment folders with their specific configuration items.
 63:         $path = $this->getPath($namespace);
 64: 
 65:         if (is_null($path))
 66:         {
 67:             return $items;
 68:         }
 69: 
 70:         // First we'll get the main configuration file for the groups. Once we have
 71:         // that we can check for any environment specific files, which will get
 72:         // merged on top of the main arrays to make the environments cascade.
 73:         $file = "{$path}/{$group}.php";
 74: 
 75:         if ($this->files->exists($file))
 76:         {
 77:             $items = $this->files->getRequire($file);
 78:         }
 79: 
 80:         // Finally we're ready to check for the environment specific configuration
 81:         // file which will be merged on top of the main arrays so that they get
 82:         // precedence over them if we are currently in an environments setup.
 83:         $file = "{$path}/{$environment}/{$group}.php";
 84: 
 85:         if ($this->files->exists($file))
 86:         {
 87:             $items = $this->mergeEnvironment($items, $file);
 88:         }
 89: 
 90:         return $items;
 91:     }

108:      * @param  string  $group
109:      * @param  string  $namespace
110:      * @return bool
111:      */
112:     public function exists($group, $namespace = null)
113:     {
114:         $key = $group.$namespace;
115: 
116:         // We'll first check to see if we have determined if this namespace and
117:         // group combination have been checked before. If they have, we will
118:         // just return the cached result so we don't have to hit the disk.
119:         if (isset($this->exists[$key]))
120:         {
121:             return $this->exists[$key];
122:         }
123: 
124:         $path = $this->getPath($namespace);
125: 
126:         // To check if a group exists, we will simply get the path based on the
127:         // namespace, and then check to see if this files exists within that
128:         // namespace. False is returned if no path exists for a namespace.
129:         if (is_null($path))
130:         {
131:             return $this->exists[$key] = false;
132:         }
133: 
134:         $file = "{$path}/{$group}.php";
135: 
136:         // Finally, we can simply check if this file exists. We will also cache
137:         // the value in an array so we don't have to go through this process
138:         // again on subsequent checks for the existing of the config file.
139:         $exists = $this->files->exists($file);
140: 
141:         return $this->exists[$key] = $exists;
142:     }

144:     /**
145:      * Apply any cascades to an array of package options.
146:      *
147:      * @param  string  $env
148:      * @param  string  $package
149:      * @param  string  $group
150:      * @param  array   $items
151:      * @return array
152:      */
153:     public function cascadePackage($env, $package, $group, $items)
154:     {
155:         // First we will look for a configuration file in the packages configuration
156:         // folder. If it exists, we will load it and merge it with these original
157:         // options so that we will easily "cascade" a package's configurations.
158:         $file = "packages/{$package}/{$group}.php";
159: 
160:         if ($this->files->exists($path = $this->defaultPath.'/'.$file))
161:         {
162:             $items = array_merge($items, $this->getRequire($path));
163:         }
164: 
165:         // Once we have merged the regular package configuration we need to look for
166:         // an environment specific configuration file. If one exists, we will get
167:         // the contents and merge them on top of this array of options we have.
168:         $path = $this->getPackagePath($env, $package, $group);
169: 
170:         if ($this->files->exists($path))
171:         {
172:             $items = array_merge($items, $this->getRequire($path));
173:         }
174:
