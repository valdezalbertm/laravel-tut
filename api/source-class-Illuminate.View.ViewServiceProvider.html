119: 
120:             $finder = $app['view.finder'];
121: 
122:             $env = new Environment($resolver, $finder, $app['events']);
123: 
124:             // We will also set the container instance on this view environment since the
125:             // view composers may be classes registered in the container, which allows
126:             // for great testable, flexible composers for the application developer.
127:             $env->setContainer($app);

129:             $env->share('app', $app);

136:      * Register the session binder for the view environment.
137:      *
138:      * @return void
139:      */
140:     protected function registerSessionBinder()
141:     {
142:         list($app, $me) = array($this->app, $this);
143: 
144:         $app->booted(function() use ($app, $me)
145:         {
146:             // If the current session has an "errors" variable bound to it, we will share
147:             // its value with all view instances so the views can easily access errors
148:             // without having to bind. An empty bag is set when there aren't errors.
149:             if ($me->sessionHasErrors($app))
150:             {
151:                 $errors = $app['session']->get('errors');
152: 
153:                 $app['view']->share('errors', $errors);
154:             }
155: 
156:             // Putting the errors in the view for every view allows the developer to just
157:             // assume that some errors are always available, which is convenient since
158:             // they don't have to continually run checks for the presence of errors.
159:             else
160:             {
161:                 $app['view']->share('errors', new MessageBag);
162:             }
163:         });
164:     }
