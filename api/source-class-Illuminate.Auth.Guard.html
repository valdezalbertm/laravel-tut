107:     public function user()
108:     {
109:         if ($this->loggedOut) return;
110: 
111:         // If we have already retrieved the user for the current request we can just
112:         // return it back immediately. We do not want to pull the user data every
113:         // request into the method becaue that would tremendously slow the app.
114:         if ( ! is_null($this->user))
115:         {
116:             return $this->user;
117:         }
118: 
119:         $id = $this->session->get($this->getName());
120: 
121:         // First we will try to load the user using the identifier in the session if
122:         // one exists. Otherwise we will check for a "remember me" cookie in this
123:         // request, and if one exists, attempt to retrieve the user using that.
124:         $user = null;
125: 
126:         if ( ! is_null($id))
127:         {
128:             $user = $this->provider->retrieveByID($id);
129:         }
130: 
131:         // If the user is null, but we decrypt a "recaller" cookie we can attempt to
132:         // pull the user data on that cookie which serves as a remember cookie on
133:         // the application. Once we have a user we can return it to the caller.
134:         $recaller = $this->getRecaller();

136:         if (is_null($user) and ! is_null($recaller))
137:         {
138:             $user = $this->provider->retrieveByID($recaller);
139:         }
140: 
141:         return $this->user = $user;
142:     }
