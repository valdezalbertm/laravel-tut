 35:      * @param  string  $key
 36:      * @return mixed
 37:      */
 38:     public function get($key)
 39:     {
 40:         $path = $this->path($key);
 41: 
 42:         // If the file doesn't exists, we obviously can't return the cache so we will
 43:         // just return null. Otherwise, we'll get the contents of the file and get
 44:         // the expiration UNIX timestamps from the start of the file's contents.
 45:         if ( ! $this->files->exists($path))
 46:         {
 47:             return null;
 48:         }
 49: 
 50:         $expire = substr($contents = $this->files->get($path), 0, 10);
 51: 
 52:         // If the current time is greater than expiration timestamps we will delete
 53:         // the file and return null. This helps clean up the old files and keeps
 54:         // this directory much cleaner for us as old files aren't hanging out.
 55:         if (time() >= $expire)
 56:         {
 57:             return $this->forget($key);
 58:         }
 59: 
 60:         return unserialize(substr($contents, 10));
 61:     }
